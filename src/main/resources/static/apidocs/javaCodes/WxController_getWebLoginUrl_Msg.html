public class Msg implements Serializable{

    private int code; //
    private Object data; //
    private String msg; //

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="data")
    public Object getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(Object data){
        this.data = data;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

}
